- name: Clone nmstate repository
  ansible.builtin.git:
    repo: https://github.com/nmstate/nmstate.git
    dest: "{{ ansible_env.HOME }}/nmstate/nmstate"
    force: yes
  when: nmstate_check.rc != 0

- name: Build and install nmstate
  ansible.builtin.shell:
    cmd: make install
    chdir: "{{ ansible_env.HOME }}/nmstate/nmstate"
  when: nmstate_check.rc != 0
  become: yes
  register: nmstate_install
  async: 1800
  poll: 10

- name: Verify nmstate installation
  ansible.builtin.command:
    argv:
      - which
      - nmstatectl
  register: nmstate_verify
  changed_when: false


- name: Display nmstate installation result
  ansible.builtin.debug:
  msg: "nmstate Installation: {{ 'SUCCESSFUL' if nmstate_verify.rc == 0 else 'FAILED' }}"
  when: nmstate_check.rc != 0

# Check nmstate dependencies
- name: Check if Rust is installed (nmstate dependency)
  ansible.builtin.command:
  argv:
    - which
    - rustc
  register: rust_check
  failed_when: false
  changed_when: false


- name: Check if git is installed (nmstate dependency)
  ansible.builtin.command:
  argv:
    - which
    - git
  register: git_check
  failed_when: false
  changed_when: false


- name: Check if build-essential is installed (nmstate dependency)
  ansible.builtin.command:
  argv:
    - dpkg
    - -l
    - build-essential
  register: build_essential_check
  failed_when: false
  changed_when: false


- name: Display nmstate dependencies check result
  ansible.builtin.debug:
  msg:
    - "nmstate Dependencies:"
    - "  Rust: {{ 'INSTALLED' if rust_check.rc is defined and rust_check.rc == 0 else 'NOT INSTALLED' }}"
    - "  Git: {{ 'INSTALLED' if git_check.rc is defined and git_check.rc == 0 else 'NOT INSTALLED' }}"
    - "  build-essential: {{ 'INSTALLED' if build_essential_check.rc is defined and build_essential_check.rc == 0 else 'NOT INSTALLED' }}"
  rescue:
  - name: nmstate check failed
  ansible.builtin.debug:
  msg: "WARNING: nmstate prerequisite check failed. Please review the errors above."
