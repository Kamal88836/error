#!/usr/bin/env python3

import sys

file_path = "proc/net/dev"

try:
    f = open(file_path)
except:
    print("Error: proc/net/dev not found. Run script inside sosreport directory.")
    sys.exit(1)

print("{:<15} {:>12} {:>12} {:>10} {:>10}".format(
    "Interface", "RxPackets", "RxErrors", "RxDrop", "TxDrop"))
print("="*60)

for line in f.readlines()[2:]:
    parts = line.split()

    iface = parts[0].replace(":", "")
    rx_packets = int(parts[2])
    rx_errs = int(parts[3])
    rx_drop = int(parts[4])
    tx_drop = int(parts[12])

    if rx_drop > 0:
        print("{:<15} {:>12} {:>12} {:>10} {:>10}".format(
            iface, rx_packets, rx_errs, rx_drop, tx_drop))
