Creating an ignition with hosts.conf added

Creating an ignition with hosts.conf added Collapse source


[root@host ~]# python3 -c "import json, sys;
ignition = json.load(sys.stdin);
storage = ignition.get('storage', {});
files = storage.get('files', []);
files.append({'path': '/etc/hosts', 'mode': 420, 'overwrite': True, 'contents': {'source': 'data:text/plain;charset=utf-8;base64,$(cat hosts.conf)'}});
storage['files'] = files;
ignition['storage'] = storage;
json.dump(ignition, sys.stdout)" < org.ign > new.ign
